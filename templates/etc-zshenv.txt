export TIMER_ALL_THEN2=$(/bin/date +%s%N) 2>/dev/null
export AHMYZSH="/ahmyzsh"
S1="${AHMYZSH}/MAIN.sh"
if [ -f "${S1}" ]; then
  . ${S1}
  unset -v S1
  source "${AHMYZSH}/MAIN.sh"
else
  if [[ -o interactive ]]; then
    echo "Error loading '${S1}'... AHMYZSH initialisation scripts can not be resolved"
  fi
  unset -v S1
fi
